1. Overview

This guide explains how to configure the Alchemy Build System to cross-compile applications for the Thundercomm Dragonwing QCS6490 development board using the Qualcomm Yocto extensible SDK (eSDK).

  • SoC: Qualcomm QCS6490

4.2 Product Configuration Files

üìÑ products/dragonwing-qcs6490/product.mk

###############################################################################
## Product setup for Dragonwing QCS6490 board
###############################################################################

# Product identification
TARGET_PRODUCT := dragonwing-qcs6490
TARGET_PRODUCT_VARIANT := aarch64

# Target OS configuration
# Note: Using empty flavour to avoid loading default yocto setup
# The Qualcomm eSDK is configured manually in toolchain-setup.mk
TARGET_OS := linux
TARGET_OS_FLAVOUR :=

# Target architecture (ARM 64-bit)
TARGET_ARCH := aarch64
TARGET_CPU := cortex-a78

# C library (glibc from Yocto SDK)
TARGET_LIBC := glibc

# Board-specific defines
TARGET_GLOBAL_CFLAGS += \
    -DTARGET_BOARD_QCS6490=1 \
    -DTARGET_BOARD_DRAGONWING=1

# Image format for deployment
TARGET_IMAGE_FORMAT := tar.gz

üìÑ products/dragonwing-qcs6490/toolchain-setup.mk

###############################################################################
## Toolchain setup for Qualcomm Yocto eSDK
###############################################################################

# SDK path (can be overridden via QCOM_SDK_PATH environment variable)
QCOM_SDK_PATH ?= $(HOME)/qcom-wayland_sdk

# Verify SDK exists
ifeq ("$(wildcard $(QCOM_SDK_PATH))","")
  $(error Qualcomm SDK not found at $(QCOM_SDK_PATH))
endif

# SDK sysroots
QCOM_SDK_TARGET_SYSROOT := $(QCOM_SDK_PATH)/tmp/sysroots/qcm6490-idp
QCOM_SDK_HOST_SYSROOT   := $(QCOM_SDK_PATH)/tmp/sysroots/x86_64

# Cross-compiler prefix (full paths)
QCOM_TOOLCHAIN_PATH := $(QCOM_SDK_HOST_SYSROOT)/usr/bin/aarch64-qcom-linux
TARGET_CROSS := $(QCOM_TOOLCHAIN_PATH)/aarch64-qcom-linux-

# Define toolchain triplet directly to avoid auto-detection issues
TARGET_TOOLCHAIN_TRIPLET := aarch64-qcom-linux

# Compiler tools (full paths)
TARGET_CC      := $(TARGET_CROSS)gcc
TARGET_CXX     := $(TARGET_CROSS)g++
TARGET_AS      := $(TARGET_CROSS)as
TARGET_LD      := $(TARGET_CROSS)ld
TARGET_AR      := $(TARGET_CROSS)ar
TARGET_STRIP   := $(TARGET_CROSS)strip
TARGET_OBJCOPY := $(TARGET_CROSS)objcopy

# Architecture flags
QCOM_ARCH_FLAGS := -march=armv8.2-a+crypto -mbranch-protection=standard

# Security hardening flags
QCOM_SECURITY_FLAGS := -fstack-protector-strong -D_FORTIFY_SOURCE=2

# Sysroot flag
QCOM_SYSROOT_FLAG := --sysroot=$(QCOM_SDK_TARGET_SYSROOT)

# Global flags
TARGET_GLOBAL_CFLAGS += $(QCOM_ARCH_FLAGS) $(QCOM_SECURITY_FLAGS) $(QCOM_SYSROOT_FLAG)
TARGET_GLOBAL_LDFLAGS += -Wl,-O1 -Wl,--as-needed $(QCOM_SYSROOT_FLAG)

# Include paths
TARGET_GLOBAL_C_INCLUDES += $(QCOM_SDK_TARGET_SYSROOT)/usr/include

5. Hello World Example

5.1 Module Definition (atom.mk)

###############################################################################
## packages/helloworld/atom.mk
###############################################################################

# Get the directory where this atom.mk file is located
LOCAL_PATH := $(call my-dir)

# Clear all LOCAL_* variables
include $(CLEAR_VARS)

# Module name (will be the executable name)
LOCAL_MODULE := helloworld

# Description
LOCAL_DESCRIPTION := Hello World example for QCS6490 board

# Source files
LOCAL_SRC_FILES := src/helloworld.c

# Compilation flags
LOCAL_CFLAGS := -Wall -Wextra -Wpedantic

# Build as executable
include $(BUILD_EXECUTABLE)

5.2 Source Code (helloworld.c)

/**
 * @file helloworld.c
 * @brief Hello World for Dragonwing QCS6490
 */

#include <stdio.h>
#include <stdlib.h>
#include <sys/utsname.h>

int main(int argc, char *argv[])
{
    struct utsname sys_info;

    printf("============================================\n");
    printf("    Hello World from Dragonwing QCS6490!\n");
    printf("============================================\n\n");

    if (uname(&sys_info) == 0) {
        printf("System Information:\n");
        printf("  OS:       %s\n", sys_info.sysname);
        printf("  Machine:  %s\n", sys_info.machine);
        printf("  Release:  %s\n", sys_info.release);
    }

    printf("\nBuild Information:\n");
    printf("  Compiled: %s %s\n", __DATE__, __TIME__);
#ifdef TARGET_BOARD_QCS6490
    printf("  Target:   Qualcomm QCS6490\n");
#endif
#if defined(__aarch64__)
    printf("  Arch:     AArch64 (64-bit ARM)\n");
#endif

    return EXIT_SUCCESS;
}

‚ö†Ô∏è Important: Workspace Structure

Alchemy requires a separate workspace directory containing your packages. The Alchemy build system itself should be in a separate location. The build is run FROM the workspace directory.

1

Create Workspace Directory

Create a workspace directory and copy the helloworld package into it:

# Create workspace structure
mkdir -p ~/workspace-qcs6490/packages

# Copy the helloworld package to your workspace
cp -r /path/to/alchemy/packages/helloworld ~/workspace-qcs6490/packages/

# Verify structure
tree ~/workspace-qcs6490/
# workspace-qcs6490/
# ‚îî‚îÄ‚îÄ packages/

2

Set Environment Variables

Configure the required environment variables:

# Navigate to YOUR WORKSPACE (not alchemy directory!)
cd ~/workspace-qcs6490

# Set ALCHEMY_HOME to point to the Alchemy build system
export ALCHEMY_HOME=/path/to/alchemy

# Set TARGET_CONFIG_DIR to point to the product configuration
export ALCHEMY_TARGET_CONFIG_DIR=$ALCHEMY_HOME/products/dragonwing-qcs6490

Using the alchemake Script

Alchemy provides a convenient alchemake wrapper script that simplifies the build command. This script handles error detection, parallel job management, and provides a cleaner output.

üõ†Ô∏è Method 1: Using ALCHEMY_WORKSPACE_DIR (Recommended)

With ALCHEMY_WORKSPACE_DIR set, you can run alchemake from anywhere:

# Set all three required environment variables
export ALCHEMY_HOME=/var/PARROT-ALCHEMY/ALCHEMY/alchemy
export ALCHEMY_WORKSPACE_DIR=$HOME/workspace-qcs6490
export ALCHEMY_TARGET_CONFIG_DIR=$ALCHEMY_HOME/products/dragonwing-qcs6490

# Now you can build from anywhere!
$ALCHEMY_HOME/scripts/alchemake helloworld

# Or build all modules
$ALCHEMY_HOME/scripts/alchemake all

üõ†Ô∏è Method 2: From Workspace Directory

Without ALCHEMY_WORKSPACE_DIR, you must be in the workspace directory and use -f:

# Navigate to workspace
cd ~/workspace-qcs6490

# Set environment variables
export ALCHEMY_HOME=/var/PARROT-ALCHEMY/ALCHEMY/alchemy
export ALCHEMY_TARGET_CONFIG_DIR=$ALCHEMY_HOME/products/dragonwing-qcs6490

# Build with -f flag
$ALCHEMY_HOME/scripts/alchemake -f $ALCHEMY_HOME/main.mk helloworld

üí° Permanent Environment Setup

Create an environment script to easily load settings:

# Create environment setup script
cat > ~/qcs6490-env.sh << 'EOF'
#!/bin/bash

# Now simply use:

7. Building Multiple Packages

Alchemy provides several mechanisms to build multiple packages together. Understanding these methods helps you organize your project efficiently.

Method 1: Command Line - Multiple Modules

üéØ Direct Module List

Specify multiple module names on the command line. Each module and its dependencies will be built:

# Build multiple modules at once
$ALCHEMY_HOME/scripts/alchemake helloworld libgreeting greeter

# Or with make directly
make -f $ALCHEMY_HOME/main.mk helloworld libgreeting greeter

Use case: Quick builds of specific modules during development.

Method 2: Automatic Dependencies with LOCAL_LIBRARIES

üîó Dependency Chain

When a module declares dependencies via LOCAL_LIBRARIES, those dependencies are automatically built first:

# In greeter/atom.mk
LOCAL_PATH := $(call my-dir)

include $(CLEAR_VARS)

LOCAL_MODULE := greeter
LOCAL_DESCRIPTION := Greeter application

LOCAL_SRC_FILES := src/main.c

# This dependency is built automatically!
LOCAL_LIBRARIES := libgreeting

include $(BUILD_EXECUTABLE)

When you build greeter, Alchemy automatically:

  1. Detects libgreeting as a dependency
  2. Builds libgreeting first (compiles, links .so)
  3. Copies headers to staging for greeter to use
  4. Then builds greeter linked against libgreeting.so

Method 3: META_PACKAGE - Group Multiple Modules

üì¶ Module Grouping

A META_PACKAGE is a virtual module that groups multiple modules together. It doesn't produce any binary but serves as a convenient way to build a collection of modules:

# In dragonwing-examples/atom.mk
LOCAL_PATH := $(call my-dir)

include $(CLEAR_VARS)

LOCAL_MODULE := dragonwing-examples
LOCAL_DESCRIPTION := All Dragonwing QCS6490 example packages
LOCAL_CATEGORY_PATH := examples

# List all modules to build with this meta-package
LOCAL_DEPENDS_MODULES := \
    helloworld \
    libgreeting \
    greeter

include $(BUILD_META_PACKAGE)

Now build everything with a single command:

# Build all grouped modules
$ALCHEMY_HOME/scripts/alchemake dragonwing-examples

# This builds: helloworld, libgreeting, greeter

Method 4: Build All Configured Modules

üåê The 'all' Target

Build every module that is enabled in the configuration:

# Build ALL modules
$ALCHEMY_HOME/scripts/alchemake all

Note: This builds all modules found in the workspace that pass configuration checks.

8. Troubleshooting

‚ùå No rule to make target 'helloworld'

Error: make: *** No rule to make target 'helloworld'. Stop.

Cause: Running make from wrong directory or packages not found

Solution:

  • Run make from workspace directory, not alchemy directory
  • Verify packages exist: ls workspace/packages/helloworld/atom.mk
  • Check ALCHEMY_HOME points to alchemy directory

‚ùå Unrecognized option '-m64' or '-msse'

Error: aarch64-qcom-linux-gcc: error: unrecognized command-line option '-m64'

Cause: x86 flags being passed to ARM cross-compiler

Solution:

  • Ensure product.mk exists (not just setup.mk)
  • Set TARGET_OS_FLAVOUR := (empty) to avoid loading default yocto setup
  • Add TARGET_TOOLCHAIN_TRIPLET := aarch64-qcom-linux in toolchain-setup.mk

‚ùå No Yocto SDK found

Error: No Yocto SDK found, you need to set TARGET_YOCTO_SDK

Cause: TARGET_OS_FLAVOUR := yocto loads default yocto setup

Solution:

  • Change TARGET_OS_FLAVOUR := yocto to TARGET_OS_FLAVOUR := (empty)
  • The Qualcomm eSDK is handled manually in toolchain-setup.mk

‚ùå TARGET_PRODUCT shows 'linux-native'

Error: Build output shows TARGET_PRODUCT = linux-native instead of dragonwing-qcs6490

Cause: Product configuration not loaded

Solution:

  • Alchemy expects product.mk or target-setup.mk
  • Create/copy: cp setup.mk product.mk
  • Verify path: ALCHEMY_TARGET_CONFIG_DIR must be absolute path

‚ùå SDK Not Found

Error: Qualcomm SDK not found at $(QCOM_SDK_PATH)

Solution:

  • Verify SDK is installed at ~/qcom-wayland_sdk/
  • Or set export QCOM_SDK_PATH=/your/sdk/path

‚ùå Cross-compiler Not Found

Error: aarch64-qcom-linux-gcc: command not found

Solution:

  • Check toolchain path exists: ls ~/qcom-wayland_sdk/tmp/sysroots/x86_64/usr/bin/aarch64-qcom-linux/
  • Ensure SDK was fully extracted

‚ùå Missing Headers

Error: fatal error: stdio.h: No such file

Solution:

  • Verify sysroot exists: ls ~/qcom-wayland_sdk/tmp/sysroots/qcm6490-idp/usr/include/
  • Check --sysroot flag in CFLAGS

‚ùå Wrong Architecture

Error: Binary doesn't run on target

Solution:

  • Verify with file helloworld
  • Should show: ELF 64-bit LSB ... ARM aarch64
  • Check TARGET_ARCH is set to aarch64
